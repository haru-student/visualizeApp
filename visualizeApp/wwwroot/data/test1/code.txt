using System;

class Test1
{
    static void Main()
    {
        int[] d = { 100, 200 };
        int p1 = 4500;
        int p2 = 1;

        int r1 = f1(p1, p2);
        int r2 = f2(d, r1);
    }
    static int f1(int p1, int p2)
    {
        int f = 1;
        if (p1 >= 2000)              
        {
            if (p1 < 5000)        
            {
                f = 2;
                if (p2 > 0) 
                    f = 3;    
            }
            else if (p1 > 8000)    
            {
                f = 5;
                if (p2 > 0)
                {
                    if (p2 < 10) 
                        f = 7; 
                }
            }
            else
                f = 4;        
        }
        return f;
    }
    static int f2(int[] d, int r1)
    {
        int sum = 0;
        for (int i = 0; i < d.Length; i++) 
        {
            int val = d[i];
            if (val > 150) 
                sum += val * r1; 
            else if (val > 50)
                sum += 50;   
            else
            sum += 10;                  
        }
        return sum;
    }
}
